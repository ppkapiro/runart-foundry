# Comando para crear/actualizar whitelist canario (CANARY_ROLE_RESOLVER_EMAILS)
# Sustituir <NAMESPACE_ID_RUNART_ROLES_PREVIEW> y correos reales.
# Formato esperado: JSON array plano de strings (emails normalizados lowercase).

wrangler kv key put --namespace-id=<NAMESPACE_ID_RUNART_ROLES_PREVIEW> \
  --key=CANARY_ROLE_RESOLVER_EMAILS \
  --value='["owner@example.com","team@example.com","client_admin@example.com","client@example.com"]' \
  --preview

# Verificar:
wrangler kv key get --namespace-id=<NAMESPACE_ID_RUNART_ROLES_PREVIEW> --key=CANARY_ROLE_RESOLVER_EMAILS --preview
